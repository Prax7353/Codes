#include <ESP8266WiFi.h>
#include <PubSubClient.h>

// WiFi credentials
const char* ssid = "YOUR_WIFI_NAME";
const char* password = "YOUR_WIFI_PASSWORD";

// MQTT broker (Mosquitto server IP)
const char* mqtt_server = "192.168.1.100";   // change to your broker IP

WiFiClient espClient;
PubSubClient client(espClient);

void setup_wifi() {
  delay(10);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
  }
}

void reconnect() {
  while (!client.connected()) {
    if (client.connect("ESP8266Publisher")) {
      // Connected
    } else {
      delay(5000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  setup_wifi();
  client.setServer(mqtt_server, 1883);  // default MQTT port
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();

  // Publish message
  String message = "Hello from ESP8266!";
  client.publish("test/topic", message.c_str());

  delay(2000); // publish every 2 sec
}
